codigo para registrar cliente
import { supabase } from "@/lib/supabaseClient";

export async function crearCliente({
  tipo_documento,
  identificacion,
  dv,
  es_juridica,
  razon_social,
  regimen,
  nombre_comercial,
  ciudad,
  direccion,
  telefono,
  telefono_contacto,
  nombre_contacto,
  correo_electronico,
  comentarios,
}: {
  tipo_documento?: string;
  identificacion: string;
  dv?: string;
  es_juridica?: boolean;
  razon_social?: string;
  regimen?: string;
  nombre_comercial?: string;
  ciudad?: string;
  direccion?: string;
  telefono?: string;
  telefono_contacto?: string;
  nombre_contacto?: string;
  correo_electronico?: string;
  comentarios?: string;
}) {
  const { data, error } = await supabase.from("clientes").insert([
    {
      tipo_documento,
      identificacion,
      dv,
      es_juridica,
      razon_social,
      regimen,
      nombre_comercial,
      ciudad,
      direccion,
      telefono,
      telefono_contacto,
      nombre_contacto,
      correo_electronico,
      comentarios,
    },
  ]);

  if (error) {
    console.error("❌ Error al crear cliente:", error);
    throw error;
  }

  return data;
}
